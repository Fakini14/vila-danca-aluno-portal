  const cleanCPF = cpf.replace(/\D/g, '')
  
  if (cleanCPF.length !== 11 || /^(\d)\1{10}$/.test(cleanCPF)) {
    return false

  let sum = 0
  for (let i = 0; i < 9; i++) {

  let remainder = (sum * 10) % 11
  if (remainder === 10 || remainder === 11) remainder = 0

  sum = 0

    sum += parseInt(cleanCPF.charAt(i)) * (11 - i)
